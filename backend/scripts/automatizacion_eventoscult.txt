const cron = require('node-cron');
const { exec } = require('child_process');

  exec('node backend/scripts/syncEventosCulturales.js', (err, stdout, stderr) => {
    if (err) {
      console.error(`Error al ejecutar sincronizaci√≥n: ${err}`);
      return;
    }
    console.log(stdout);
  });
});
